📖 วิธีเชื่อม Google Sheets
ทำครั้งเดียว ใช้เวลา ~3 นาที
ขั้นที่ 1 — เปิด sheets.new สร้าง Google Sheet ใหม่ ตั้งชื่ออะไรก็ได้
ขั้นที่ 2 — เมนูบนสุด กด Extensions → Apps Script
ขั้นที่ 3 — ลบโค้ดเดิมทั้งหมด แล้ววางโค้ดนี้ลงไป: 
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);
  writeTab(sheet, "รายรับ", data.revenues, ["date","doctorId","amount","note"]);
  writeTab(sheet, "รายจ่าย", data.expenses, ["date","category","amount","doctorId","note"]);
  writeTab(sheet, "ประกันสังคม", data.socialSecurity, ["date","amount","note"]);
  writeTab(sheet, "หมอ", data.doctors, ["id","name","color"]);
  return ContentService.createTextOutput(
    JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}
function writeTab(ss, name, rows, cols) {
  var sh = ss.getSheetByName(name) || ss.insertSheet(name);
  sh.clear();
  sh.appendRow(cols);
  (rows||[]).forEach(function(r){
    sh.appendRow(cols.map(function(c){ return r[c]!==undefined ? r[c] : ""; }));
  });
}
ขั้นที่ 4 — กด Deploy → New deployment → ไอคอนเฟือง เลือก Web app
ขั้นที่ 5 — ตั้งค่า:
• Execute as: Me
• Who has access: Anyone
แล้วกด Deploy (ครั้งแรกต้องกด Authorize อนุญาตสิทธิ์)
ขั้นที่ 6 — คัดลอก Web app URL ที่ได้ (ลงท้ายด้วย /exec) → กลับมาวางในช่องเชื่อมต่อ แล้วกด "เชื่อมต่อ"
💡 หมายเหตุ: ถ้าแก้โค้ด Apps Script ภายหลัง ต้อง Deploy ใหม่ทุกครั้ง (Manage deployments → Edit → New version)
